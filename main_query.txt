query RQ01 {
  search(query: "stars:>100", type: REPOSITORY, first: 100, after: null) {
    pageInfo{
      hasNextPage
      endCursor
    }
    nodes {
      ... on Repository {
        stargazerCount
        createdAt
        open_issues: issues{
          totalCount
        }
        closed_issues: issues(states: CLOSED) {
          totalCount
        }
        name
        updatedAt
        primaryLanguage {
          name
        }
        pullRequests(states: MERGED) {
          totalCount
        }
        releases {
          totalCount
        }
        url
      }
    }
  }
}
