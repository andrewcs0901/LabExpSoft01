query RQ01 {
  search(query: "stars:>100", type: REPOSITORY, first: 100) {
    nodes {
      ... on Repository {
        createdAt
        issues{
          totalCount
        }
        name
        updatedAt
        primaryLanguage {
          name
        }
        pullRequests(states: MERGED) {
          totalCount
        }
        releases{
          totalCount
        }
      }
    }
  }
}
